<script>

  // DEMO DATA (replaced by Firebase automatically)

  const demoCars=[{

    title:'Toyota Corolla',year:2020,make:'Toyota',model:'Corolla',price:'₪72,000',desc:'Excellent condition, low mileage.',

    images:[

      'https://images.unsplash.com/photo-1617814072417-2b0a8b68b1a3',

      'https://images.unsplash.com/photo-1549924231-f129b911e442'

    ]

  }];


  const grid=document.getElementById('cars');


  demoCars.forEach(car=>{

    const card=document.createElement('div');card.className='card';

    let idx=0;

    card.innerHTML=`

      <div class="carousel">

        ${car.images.map((i,n)=>`<img src="${i}" class="${n===0?'active':''}">`).join('')}

        <button class="nav-btn prev">‹</button>

        <button class="nav-btn next">›</button>

      </div>

      <div class="thumbs">

        ${car.images.map((i,n)=>`<img src="${i}" class="${n===0?'active':''}">`).join('')}

      </div>

      <div class="card-body">

        <div class="title">${car.year} ${car.make} ${car.model}</div>

        <div class="meta">${car.desc}</div>

        <div class="price">${car.price}</div>

        <a class="wa" href="https://wa.me/972000000000">WhatsApp</a>

      </div>`;


    const imgs=[...card.querySelectorAll('.carousel img')];

    const thumbs=[...card.querySelectorAll('.thumbs img')];


    function show(i){imgs.forEach((im,n)=>im.classList.toggle('active',n===i));thumbs.forEach((t,n)=>t.classList.toggle('active',n===i));idx=i}


    card.querySelector('.prev').onclick=()=>show((idx-1+imgs.length)%imgs.length);

    card.querySelector('.next').onclick=()=>show((idx+1)%imgs.length);

    thumbs.forEach((t,n)=>t.onclick=()=>show(n));

    imgs.forEach(im=>im.onclick=()=>openLightbox(im.src));


    grid.appendChild(card);

  });


  function openLightbox(src){document.getElementById('lightboxImg').src=src;document.getElementById('lightbox').classList.add('active')}

  function closeLightbox(){document.getElementById('lightbox').classList.remove('active')}

</script>
